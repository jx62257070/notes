①.安装nodemon
npm install -g nodemon
②.添加配置
{
    "version": "0.2.0",
    "configurations": [{
        "type": "node",
        "request": "launch",
        "name": "nodemon",
        "runtimeExecutable": "nodemon",
        "program": "${workspaceRoot}\\bin\\www",
        "restart": true,
        "env": {
            "NODE_ENV": "dev"
        },
        "console": "integratedTerminal",
        "internalConsoleOptions": "neverOpen"
    }]
}
③.调试
